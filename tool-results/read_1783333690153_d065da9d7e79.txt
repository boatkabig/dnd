    10→/* ---------------- DICE ENGINE ---------------- */
    11→const d = (sides) => Math.floor(Math.random() * sides) + 1;
    12→
    13→function rollFormula(formula) {
    14→  // "2d6+3" -> { total, rolls:[..], mod }
    15→  const m = String(formula).replace(/\s/g, "").match(/^(\d*)d(\d+)([+-]\d+)?$/i);
    16→  if (!m) return { total: 0, rolls: [], mod: 0, formula };
    17→  const n = parseInt(m[1] || "1", 10);
    18→  const sides = parseInt(m[2], 10);
    19→  const mod = m[3] ? parseInt(m[3], 10) : 0;
    20→  const rolls = [];
    21→  for (let i = 0; i < n; i++) rolls.push(d(sides));
    22→  return { total: rolls.reduce((a, b) => a + b, 0) + mod, rolls, mod, formula };
    23→}
    24→
    25→function rollD20(mod, adv = "none") {
    26→  const a = d(20), b = d(20);
    27→  let die = a;
    28→  if (adv === "advantage") die = Math.max(a, b);
    29→  if (adv === "disadvantage") die = Math.min(a, b);
    30→  return { die, other: adv !== "none" ? (die === a ? b : a) : null, mod, total: die + mod, adv };
    31→}

   268→async function fetchSRDMonster(indexRaw) {
   269→  const index = String(indexRaw).toLowerCase().replace(/_/g, "-").trim();
   270→  if (srdCache[index]) return srdCache[index];
   271→  try {
   272→    const r = await fetch(`${SRD_BASE}/monsters/${index}`);
   273→    if (!r.ok) return null;
   274→    const m = await r.json();
   275→    // เลือก action โจมตีอันแรกที่มี attack roll + damage dice
   276→    let atkBonus = 3, dmg = "1d6+1";
   277→    const act = (m.actions || []).find((a) => a.attack_bonus !== undefined && Array.isArray(a.damage) && a.damage.some((d) => d.damage_dice));
   278→    if (act) {
   279→      atkBonus = act.attack_bonus;
   280→      const dd = act.damage.find((d) => d.damage_dice);
   281→      if (dd) dmg = String(dd.damage_dice).replace(/\s/g, "");
   282→    }
   283→    // save mods: เริ่มจาก ability mod แล้วทับด้วย saving-throw proficiency ถ้ามี
   284→    const sv = { dex: mod(m.dexterity), con: mod(m.constitution), wis: mod(m.wisdom) };
   285→    (m.proficiencies || []).forEach((p) => {
   286→      const n = (p.proficiency && p.proficiency.index) || "";
   287→      if (n === "saving-throw-dex") sv.dex = p.value;
   288→      if (n === "saving-throw-con") sv.con = p.value;
   289→      if (n === "saving-throw-wis") sv.wis = p.value;
   290→    });
   291→    const ac = Array.isArray(m.armor_class) ? ((m.armor_class[0] && m.armor_class[0].value) || 12) : (m.armor_class || 12);
   292→    const mon = { th: m.name, ac, hp: m.hit_points || 10, atk: atkBonus, dmg, init: mod(m.dexterity), xp: m.xp || 50, sv, cr: m.challenge_rating, srd: true };
   293→    srdCache[index] = mon;
   294→    return mon;
   295→  } catch (e) { return null; }
   296→}

   384→/* ---------------- MAP ENGINE ---------------- */
   385→const DIRV = { n: [0, -1], s: [0, 1], e: [1, 0], w: [-1, 0], ne: [1, -1], nw: [-1, -1], se: [1, 1], sw: [-1, 1] };
   386→const MAP_ICON = { town: "🏘️", building: "🏠", room: "▦", dungeon: "🕳️", wilderness: "🌲", place: "📍" };
   387→
   388→function emptyMap() { return { nodes: {}, edges: [], current: null }; }
   389→
   390→function applyMapUpdate(mu, mp, pushEntry) {
   391→  if (!mu) return mp;
   392→  const m = mp
   393→    ? { nodes: { ...mp.nodes }, edges: mp.edges.slice(), current: mp.current }
   394→    : emptyMap();
   395→  const al = mu.add_location;
   396→  if (al && al.id) {
   397→    if (!m.nodes[al.id]) {
   398→      let x = 0, y = 0;
   399→      const fromId = al.from && m.nodes[al.from] ? al.from : m.current;
   400→      if (fromId && m.nodes[fromId]) {
   401→        const v = DIRV[al.dir] || [1, 0];
   402→        x = m.nodes[fromId].x + v[0];
   403→        y = m.nodes[fromId].y + v[1];
   404→        let guard = 0;
   405→        while (Object.values(m.nodes).some((n) => n.x === x && n.y === y) && guard < 10) { x += 1; guard += 1; }
   406→        m.edges.push([fromId, al.id]);
   407→      }
   408→      m.nodes[al.id] = { name: al.name || al.id, type: MAP_ICON[al.type] ? al.type : "place", x, y };
   409→      if (pushEntry) pushEntry(`🗺️ ค้นพบสถานที่ใหม่: ${al.name || al.id}`);
   410→    } else if (m.current && m.current !== al.id && !m.edges.some(([a, b]) => (a === m.current && b === al.id) || (a === al.id && b === m.current))) {
   411→      m.edges.push([m.current, al.id]);
   412→    }
   413→  }
   414→  if (mu.connect && Array.isArray(mu.connect) && mu.connect.length === 2 && m.nodes[mu.connect[0]] && m.nodes[mu.connect[1]]) {
   415→    if (!m.edges.some(([a, b]) => (a === mu.connect[0] && b === mu.connect[1]) || (a === mu.connect[1] && b === mu.connect[0]))) m.edges.push([mu.connect[0], mu.connect[1]]);
   416→  }
   417→  if (mu.move_to && m.nodes[mu.move_to]) m.current = mu.move_to;
   418→  return m;
   419→}

   468→async function callDM(systemPrompt, history) {
   469→  const response = await fetch("https://api.anthropic.com/v1/messages", {
   470→    method: "POST",
   471→    headers: { "Content-Type": "application/json" },
   472→    body: JSON.stringify({
   473→      model: "claude-sonnet-4-6",
   474→      max_tokens: 1000,
   475→      system: systemPrompt,
   476→      messages: history,
   477→    }),
   478→  });
   479→  const data = await response.json();
   480→  const text = (data.content || [])
   481→    .filter((b) => b.type === "text")
   482→    .map((b) => b.text)
   483→    .join("\n");
   484→  const clean = text.replace(/```json|```/g, "").trim();
   485→  const start = clean.indexOf("{");
   486→  const end = clean.lastIndexOf("}");
   487→  if (start === -1 || end === -1) throw new Error("DM ตอบไม่เป็น JSON");
   488→  return JSON.parse(clean.slice(start, end + 1));
   489→}

   614→  function applyUpdates(u, cc, entries) {
   615→    if (!u) return cc;
   616→    const nc = { ...cc, conditions: [...cc.conditions], inventory: [...cc.inventory] };
   617→    if (u.hp_delta) {

   619→      entries.push(entrySystem(`HP ${u.hp_delta > 0 ? "+" : ""}${u.hp_delta} → ${nc.hp}/${nc.maxHp}`));
   620→    }
   621→    if (u.gold_delta) {
   622→      nc.gold = Math.max(0, nc.gold + u.gold_delta);
   623→      entries.push(entrySystem(`ทอง ${u.gold_delta > 0 ? "+" : ""}${u.gold_delta} → ${nc.gold} gp`));
   624→    }

   626→    (u.items_use || []).forEach((it) => {
   627→      const idx = nc.inventory.indexOf(it);
   628→      if (idx < 0) { entries.push(entrySystem(`ไม่มี ${it} ในเป้สัมภาระ`)); return; }

   636→        const ci = nc.conditions.indexOf(consum.cure);
   637→        if (ci >= 0) { nc.conditions.splice(ci, 1); entries.push(entrySystem(`🧪 ใช้ ${it}: หายจากสถานะ ${consum.cure}`)); }

   639→      } else {
   640→        entries.push(entrySystem(`ใช้: ${it}`));
   641→      }
   642→    });
   643→    (u.items_add || []).forEach((it) => { nc.inventory.push(it); entries.push(entrySystem(`ได้รับ: ${it}`)); });
   644→    (u.items_remove || []).forEach((it) => {
   645→      const i = nc.inventory.indexOf(it); if (i >= 0) { nc.inventory.splice(i, 1); entries.push(entrySystem(`เสีย: ${it}`)); }
   646→    });

   649→    if (u.xp_award) return gainXP(nc, u.xp_award, entries);

   678→  async function initCombat(monsterIds, cc, entries, surprise = false) {
   679→    const enemies = [];

   681→      const base = BESTIARY[id] || (SRD_OK ? await fetchSRDMonster(id) : null);
   682→      if (base) enemies.push({ uid: `${id}_${enemies.length}`, id, ...base, hpNow: base.hp });
   683→      else entries.push(entrySystem(`⚠️ engine ไม่รู้จักมอนสเตอร์ "${id}" — ข้ามตัวนี้`));
   684→    }
   685→    if (enemies.length === 0) return null;
   686→    const pInit = rollD20(mod(cc.abilities.dex));
   687→    const eInitBest = Math.max(...enemies.map((e) => d(20) + e.init));
   688→    entries.push({ id: nextId(), type: "roll", title: "Initiative", roll: pInit, extra: `ศัตรูทอยได้สูงสุด ${eInitBest}`, success: surprise || pInit.total >= eInitBest, dc: eInitBest });
   689→    // RAW: surprise ไม่เปลี่ยน initiative แต่ฝ่ายที่ surprised ขยับไม่ได้ในเทิร์นแรก → ผลสุทธิ = ผู้เล่นได้ลงมือก่อนเสมอ
   690→    const playerFirst = surprise ? true : pInit.total >= eInitBest;
   691→    entries.push(entrySystem(`⚔️ เข้าสู่ Combat! ${enemies.map((e) => e.th).join(", ")} — ${surprise ? "ศัตรูไม่ทันตั้งตัว!" : playerFirst ? "คุณได้เริ่มก่อน" : "ศัตรูเริ่มก่อน"}`));
   692→    if (surprise) entries.push(entrySystem("🗡️ Surprise Round! การโจมตีของคุณในรอบนี้ได้เปรียบ (unseen attacker) และศัตรูเสียเทิร์นแรก — Sneak Attack ทำงานเต็ม ๆ"));
   693→    const cb = { enemies, round: 1, playerFirst, dodge: false, surprise: !!surprise };
   694→    return cb;

   697→  function enemyAttacks(cb, cc, entries) {
   698→    let nc = { ...cc };
   699→    let uncannyUsed = false;
   700→    // 5e RAW: restrained/blinded/prone → ศัตรูโจมตีคุณแบบได้เปรียบ · Dodge → เสียเปรียบ · ทั้งคู่ = หักล้างกัน
   701→    const enemyHasAdv = nc.conditions.some((k) => ENEMY_ADV_CONDS.includes(k));
   702→    for (const e of cb.enemies) {
   703→      if (e.hpNow <= 0 || nc.hp <= 0) continue;
   704→      // RAW: Dodge / ล่องหน / ซ่อนตัวสำเร็จ (unseen target) → ศัตรูโจมตีคุณแบบเสียเปรียบ
   705→      const defAdv = cb.dodge || cb.invisible || nc.hiddenAdv;
   706→      let adv = "none";
   707→      if (enemyHasAdv && !defAdv) adv = "advantage";
   708→      else if (!enemyHasAdv && defAdv) adv = "disadvantage";
   709→      const atk = rollD20(e.atk, adv);
   710→      const hit = atk.die !== 1 && (atk.die === 20 || atk.total >= nc.ac);
   711→      let extra = "";
   712→      if (hit) {
   713→        let dmgR = rollFormula(e.dmg);
   714→        let dmg = dmgR.total;
   715→        if (atk.die === 20) { const extraR = rollFormula(e.dmg.replace(/[+-]\d+$/, "") || e.dmg); dmg += extraR.rolls.reduce((a, b) => a + b, 0); }
   716→        extra = `ดาเมจ ${e.dmg}${atk.die === 20 ? " (CRIT x2 dice)" : ""} = ${dmg}`;
   717→        if (e.poison) {
   718→          const sv = rollD20(saveMod(nc, "con"));
   719→          let psR = rollFormula(e.poison.dmg);
   720→          let ptotal = psR.total;
   721→          if (atk.die === 20) ptotal += rollFormula(e.poison.dmg).rolls.reduce((a, b) => a + b, 0); // RAW: คริทบเต๋าดาเมจทุกก้อนของการโจมตีนั้น รวมพิษ
   722→          const pdmg = sv.total >= e.poison.dc ? Math.floor(ptotal / 2) : ptotal;
   723→          extra += ` · CON save ${sv.total} vs DC ${e.poison.dc} → พิษ +${pdmg}`;
   724→          dmg += pdmg;
   725→        }
   726→        // Uncanny Dodge (Rogue Lv.5): halve first hit each round
   727→        if (hasFeature(nc, "uncanny_dodge") && !uncannyUsed) {
   728→          dmg = Math.floor(dmg / 2);
   729→          uncannyUsed = true;
   730→          extra += ` · 🌀 Uncanny Dodge ลดครึ่ง → ${dmg}`;
   731→        }
   732→        nc.hp = Math.max(0, nc.hp - dmg);
   733→        // Concentration check (RAW: เฉพาะเวทที่ใช้สมาธิ — Spirit Guardians · Spiritual Weapon ไม่ใช้สมาธิ)
   734→        if (cb.spiritGuardians && dmg > 0 && nc.hp > 0) {
   735→          const dc = Math.max(10, Math.floor(dmg / 2));
   736→          const sv = rollD20(saveMod(nc, "con"));
   737→          if (sv.total < dc) {
   738→            cb.spiritGuardians = false;
   739→            entries.push(entrySystem(`💫 เสียสมาธิ! (CON save ${sv.total} vs DC ${dc}) Spirit Guardians สลายไป`));
   740→          }
   741→        }
   742→      }
   743→      entries.push({ id: nextId(), type: "roll", title: `${e.th} โจมตี`, roll: atk, vsAc: nc.ac, success: hit, extra: hit ? extra + ` → HP คุณเหลือ ${nc.hp}` : null });
   744→      if (nc.hp <= 0) {
   745→        entries.push(entrySystem(`💀 ${nc.name} ล้มลงหมดสติ! ต้องทอย Death Saving Throw`));
   746→        break;
   747→      }
   748→    }
   749→    return nc;
   750→  }

   752→  function checkCombatEnd(cb, cc, entries) {
   753→    const alive = cb.enemies.filter((e) => e.hpNow > 0);
   754→    if (alive.length === 0) {
   755→      const xp = cb.enemies.reduce((a, e) => a + e.xp, 0);

   757→      const nc = gainXP(cc, xp, entries);
   758→      return { ended: true, cc: nc };
   759→    }
   760→    return { ended: false, cc };

   763→  async function narrateCombatEvent(summary, cc, sc, baseLog, hist) {
   764→    setThinking(true);
   765→    try {
   766→      const newHist = [...hist, { role: "user", content: summary }];
   767→      const res = await callDM(buildSystemPrompt(cc), newHist);
   768→      const entries = [entryNarration(res.narration)];
   769→      let nc = applyUpdates(res.updates, cc, entries);
   770→      let nsc = res.scene || sc;
   771→      let ncb = null;
   772→      const nmp = applyMapUpdate(res.map_update, mapRef.current, (t) => entries.push(entrySystem(t)));
   773→      mapRef.current = nmp;
   774→      if (res.start_combat && res.start_combat.monsters) {
   775→        ncb = await initCombat(res.start_combat.monsters, nc, entries, res.start_combat.surprise);
   776→      }
   777→      const finalHist = [...newHist, { role: "assistant", content: JSON.stringify(res) }];
   778→      const finalLog = [...baseLog, ...entries];
   779→      cRef.current = nc; combatRef.current = ncb; logDataRef.current = finalLog;
   780→      setC(nc); setScene(nsc); setLog(finalLog); setCombat(ncb); setHistory(finalHist); setMap(nmp);
   781→      persist(nc, nsc, finalLog, ncb, finalHist);
   782→      if (nc.dead) setPhase("dead");
   783→    } catch (e) {
   784→      const finalLog = [...baseLog, entrySystem("⚠️ DM ขัดข้อง: " + e.message + " (ลองพิมพ์ต่อได้)")];
   785→      setLog(finalLog);
   786→    } finally { setThinking(false); }
   787→  }
   788→
   789→  // commit แบบอัปเดต ref ทันที — การแตะครั้งถัดไปเห็นค่าล่าสุดเสมอแม้ React ยังไม่ re-render
   790→  function commitCombat(cc2, cb2, log2) {
   791→    cRef.current = cc2; combatRef.current = cb2; logDataRef.current = log2;
   792→    setC(cc2); setCombat(cb2); setLog(log2);
   793→  }
   794→
   795→  function playerCombatAction(kind, payload) {
   796→    const combat0 = combatRef.current;
   797→    const c0 = cRef.current;
   798→    const log0 = logDataRef.current;
   799→    if (!combat0 || thinking) return;
   800→    let cc = { ...c0 };
   801→    let cb = { ...combat0, enemies: combat0.enemies.map((e) => ({ ...e })) };
   802→    const entries = [];
   803→
   804→    // --- unconscious: death save loop ---
   805→    if (cc.hp <= 0 && !cc.dead) {
   806→      const r = rollD20(0);
   807→      let ds = { ...cc.deathSaves };
   808→      if (r.die === 20) { cc.hp = 1; ds = { s: 0, f: 0 }; entries.push({ id: nextId(), type: "roll", title: "Death Save", roll: r, success: true, extra: "Nat 20! ฟื้นขึ้นมาด้วย 1 HP" }); }
   809→      else if (r.total >= 10) { ds.s += 1; entries.push({ id: nextId(), type: "roll", title: "Death Save", roll: r, dc: 10, success: true, extra: `สำเร็จ ${ds.s}/3` }); }
   810→      else { ds.f += r.die === 1 ? 2 : 1; entries.push({ id: nextId(), type: "roll", title: "Death Save", roll: r, dc: 10, success: false, extra: `ล้มเหลว ${ds.f}/3` }); }
   811→      cc.deathSaves = ds;
   812→      if (ds.f >= 3) {
   813→        cc.dead = true;
   814→        entries.push(entrySystem(`☠️ ${cc.name} เสียชีวิต...`));
   815→        const finalLog = [...log0, ...entries];
   816→        commitCombat(cc, null, finalLog); persist(cc, scene, finalLog, null, history);
   817→        setPhase("dead");
   818→        return;
   819→      }
   820→      if (ds.s >= 3 || cc.hp > 0) {
   821→        if (cc.hp <= 0) { entries.push(entrySystem("อาการคงที่ (stable) — ศัตรูทิ้งคุณไว้และจากไป")); }
   822→        cc.deathSaves = { s: 0, f: 0 };
   823→        if (cc.hp <= 0) cc.hp = 1;
   824→        const finalLog = [...log0, ...entries];
   825→        commitCombat(cc, null, finalLog);
   826→        narrateCombatEvent(`[จบ combat] ${cc.name} หมดสติแต่รอดชีวิต (stable, 1 HP) ศัตรูจากไปแล้ว บรรยายฉากที่ฟื้นขึ้นมา`, cc, scene, finalLog, history);
   827→        return;
   828→      }
   829→      // still down: enemies wait (cinematic), next round
   830→      cb.round += 1;
   831→      const finalLog = [...log0, ...entries];
   832→      commitCombat(cc, cb, finalLog); persist(cc, scene, finalLog, cb, history);
   833→      return;
   834→    }
   835→
   836→    cb.dodge = false;
   837→    cc.conditions = [...cc.conditions];
   838→    // RAW: ลุกจาก prone ใช้ movement ครึ่งหนึ่ง — engine ให้ลุกอัตโนมัติต้นเทิร์น
   839→    const proneIdx = cc.conditions.indexOf("prone");
   840→    if (proneIdx >= 0) {
   841→      cc.conditions.splice(proneIdx, 1);
   842→      entries.push(entrySystem("🧍 ลุกขึ้นยืน (ใช้ movement ครึ่งหนึ่ง) — หายจากสถานะ Prone"));
   843→    }
   844→    let fled = false;
   845→    let endsTurn = true; // bonus actions และ Action Surge ทำให้ยังไม่จบเทิร์น
   846→
   847→    const doWeaponAttack = (w, label) => {
   848→      const target = cb.enemies.find((e) => e.uid === payload && e.hpNow > 0) || cb.enemies.find((e) => e.hpNow > 0);
   849→      if (!target || !w) return;
   850→      let adv = cc.hiddenAdv || cb.surprise || cb.invisible || target.glow ? "advantage" : "none"; // surprise/ล่องหน = unseen attacker, glow = โดน Guiding Bolt
   851→      if (hasDisadv(cc)) adv = adv === "advantage" ? "none" : "disadvantage";
   852→      const atk = rollD20(attackMod(cc, w), adv);
   853→      if (target.glow) target.glow = false; // RAW: ใช้ได้กับการโจมตีถัดไปครั้งเดียว

   855→      const hit = atk.die !== 1 && (atk.die === 20 || atk.total >= target.ac);

   876→          } else parts.push(`🐍 พิษถูกต้านทาน (CON save ${sv.total} ≥ 15)`);
   877→        }
   878→        target.hpNow = Math.max(0, target.hpNow - dmg);
   879→        extra = `${parts.join(" · ")} = ${dmg} ดาเมจ → ${target.th} ${target.hpNow <= 0 ? "ตาย!" : `เหลือ ${target.hpNow} HP`}`;
   880→      }
   881→      entries.push({ id: nextId(), type: "roll", title: `${label} ${target.th} (${w.th})`, roll: atk, vsAc: target.ac, success: hit, extra });
   882→      cc.hiddenAdv = false;
   883→      if (cb.invisible) { cb.invisible = false; entries.push(entrySystem("🫥 ร่างของคุณปรากฏขึ้นอีกครั้ง (การโจมตีทำให้การล่องหนสิ้นสุด)")); }
   884→    };
   885→
   886→    if (kind === "attack" || kind === "attack_ranged") {
   887→      const w = kind === "attack_ranged" ? getRanged(cc) : getMelee(cc);
   888→      const label = kind === "attack_ranged" ? "🏹 ยิง" : "โจมตี";
   889→      const numAttacks = hasFeature(cc, "extra_attack") ? 2 : 1;
   890→      for (let i = 0; i < numAttacks; i++) {
   891→        if (!cb.enemies.some((e) => e.hpNow > 0)) break;

   894→      }

   902→          const h = rollFormula(item.heal);

   907→          const ci = cc.conditions.indexOf(item.cure);

   910→          else entries.push(entrySystem(`🧪 ใช้ ${payload}: ไม่มีสถานะ ${item.cure} อยู่ (เสียของฟรี)`));
   911→        }

   914→          cb.bonusUsed = true; endsTurn = false;
   915→          entries.push(entrySystem("🖐️ Fast Hands (bonus action) — ยังทำแอคชันหลักต่อได้"));

   939→            // RAW: Spiritual Weapon ไม่ใช้สมาธิ อยู่ได้ 1 นาที (10 รอบ) — ใช้คู่ Spirit Guardians ได้

   941→            entries.push(entrySystem(`✨ ${sp.th}: อาวุธวิญญาณปรากฏ! โจมตีเองทุกรอบ 1d8+WIS นาน 10 รอบ (ไม่ใช้สมาธิ)`));

   960→              entries.push({ id: nextId(), type: "roll", title: `${sp.th} ลำที่ ${i + 1} → ${t.th}`, roll: atk, vsAc: t.ac, success: hit, extra });
   961→            }

   963→            const alive = cb.enemies.filter((e) => e.hpNow > 0);

   968→              if (sp.kind === "attack") {
   969→                let adv = t.glow || cb.surprise || cb.invisible ? "advantage" : "none"; // Guiding Bolt mark / Surprise / ล่องหน

  1275→    } finally {

  1277→    }

  1295→  function applyAsi() {

  1349→      }
  1350→    }
  1351→    const finalLog = [...log, ...entries];
  1352→    setC(cc); setLog(finalLog);
  1353→    persist(cc, scene, finalLog, null, history);
