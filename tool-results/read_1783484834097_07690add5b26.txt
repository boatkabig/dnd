   131‚Üí   131‚Üíasync function fetchSRD(path: string): Promise<any> {
   132‚Üí   132‚Üí  if (cache.has(path)) return cache.get(path);
   133‚Üí   133‚Üí  const r = await fetch(`/api/srd?${path}`);
   134‚Üí   134‚Üí  if (!r.ok) return null;
   135‚Üí   135‚Üí  const data = await r.json();
   136‚Üí   136‚Üí  if (data && !data.error) cache.set(path, data);
   137‚Üí   137‚Üí  return data;
   138‚Üí   138‚Üí}

   200‚Üí   200‚Üíexport function normalizeSpell(m: any, slotLevel?: number, charLevel = 1): NormalizedSpell {
   201‚Üí   201‚Üí  const damage = m.damage || {};
   202‚Üí   202‚Üí  const heal = m.heal_at_slot_level || {};
   203‚Üí   203‚Üí  const hasDamage = !!damage.damage_type && (damage.damage_at_slot_level || damage.damage_at_character_level);
   204‚Üí   204‚Üí  const hasHeal = !!heal && Object.keys(heal).length > 0;
   205‚Üí   205‚Üí  const hasSave = !!m.save;
   206‚Üí   206‚Üí  const hasAttack = !!m.attack_type;
   207‚Üí   207‚Üí  const isAuto = hasDamage && !hasSave && !hasAttack; // e.g. Magic Missile (auto-hit, no save)
   208‚Üí   208‚Üí  const concentration = !!m.concentration;
   209‚Üí   209‚Üí  const bonusAction = (m.casting_time || "").toLowerCase().includes("bonus action");
   210‚Üí   210‚Üí
   211‚Üí   211‚Üí  // Determine damage formula
   212‚Üí   212‚Üí  let dmgFormula: string | undefined;
   213‚Üí   213‚Üí  let damageScaling: "slot" | "character" | "none" = "none";
   214‚Üí   214‚Üí  if (damage.damage_at_slot_level && slotLevel) {
   215‚Üí   215‚Üí    // Pick the highest level ‚â§ slotLevel
   216‚Üí   216‚Üí    const levels = Object.keys(damage.damage_at_slot_level).map(Number).sort((a, b) => a - b);
   217‚Üí   217‚Üí    const best = levels.filter((l) => l <= slotLevel).pop() || levels[0];
   218‚Üí   218‚Üí    dmgFormula = damage.damage_at_slot_level[best];
   219‚Üí   219‚Üí    damageScaling = "slot";
   220‚Üí   220‚Üí  } else if (damage.damage_at_character_level) {
   221‚Üí   221‚Üí    const levels = Object.keys(damage.damage_at_character_level).map(Number).sort((a, b) => a - b);
   222‚Üí   222‚Üí    const best = levels.filter((l) => l <= charLevel).pop() || levels[0];
   223‚Üí   223‚Üí    dmgFormula = damage.damage_at_character_level[best];
   224‚Üí   224‚Üí    damageScaling = "character";
   225‚Üí   225‚Üí  }
   226‚Üí   226‚Üí
   227‚Üí   227‚Üí  // Determine heal formula
   228‚Üí   228‚Üí  let healFormula: string | undefined;
   229‚Üí   229‚Üí  if (hasHeal && slotLevel) {
   230‚Üí   230‚Üí    const levels = Object.keys(heal).map(Number).sort((a, b) => a - b);
   231‚Üí   231‚Üí    const best = levels.filter((l) => l <= slotLevel).pop() || levels[0];
   232‚Üí   232‚Üí    healFormula = heal[best];
   233‚Üí   233‚Üí  }
   234‚Üí   234‚Üí
   235‚Üí   235‚Üí  // Determine spell kind
   236‚Üí   236‚Üí  let kind: NormalizedSpell["kind"] = "utility";
   237‚Üí   237‚Üí  if (hasHeal) kind = "heal";
   238‚Üí   238‚Üí  else if (hasAttack && hasDamage) kind = "attack";
   239‚Üí   239‚Üí  else if (hasSave && hasDamage) kind = "save";
   240‚Üí   240‚Üí  else if (isAuto) kind = "auto";
   241‚Üí   241‚Üí  else if (concentration || (m.duration || "").toLowerCase().includes("minute") || (m.duration || "").toLowerCase().includes("hour")) kind = "buff";
   242‚Üí   242‚Üí
   243‚Üí   243‚Üí  // Conditions applied (parsed from description ‚Äî basic heuristic for common buff/debuff spells)
   244‚Üí   244‚Üí  const conditionsAdd: string[] = [];
   245‚Üí   245‚Üí  const descLower = (m.desc || []).join(" ").toLowerCase();
   246‚Üí   246‚Üí  const conditionKeywords: Record<string, string[]> = {
   247‚Üí   247‚Üí    poisoned: ["poisoned"],
   248‚Üí   248‚Üí    frightened: ["frightened"],
   249‚Üí   249‚Üí    restrained: ["restrained"],
   250‚Üí   250‚Üí    blinded: ["blinded"],
   251‚Üí   251‚Üí    charmed: ["charmed"],
   252‚Üí   252‚Üí    prone: ["knocked prone", "fall prone"],
   253‚Üí   253‚Üí  };
   254‚Üí   254‚Üí  for (const [cond, keywords] of Object.entries(conditionKeywords)) {
   255‚Üí   255‚Üí    if (keywords.some((kw) => descLower.includes(kw))) conditionsAdd.push(cond);
   256‚Üí   256‚Üí  }
   257‚Üí   257‚Üí
   258‚Üí   258‚Üí  return {
   259‚Üí   259‚Üí    index: m.index,
   260‚Üí   260‚Üí    name: m.name,
   261‚Üí   261‚Üí    level: m.level,
   262‚Üí   262‚Üí    school: m.school?.name || "Unknown",
   263‚Üí   263‚Üí    casting_time: m.casting_time || "1 action",
   264‚Üí   264‚Üí    range: m.range || "Self",
   265‚Üí   265‚Üí    duration: m.duration || "Instantaneous",
   266‚Üí   266‚Üí    concentration,
   267‚Üí   267‚Üí    ritual: !!m.ritual,
   268‚Üí   268‚Üí    components: m.components || [],
   269‚Üí   269‚Üí    material: m.material,
   270‚Üí   270‚Üí    desc: Array.isArray(m.desc) ? m.desc.join(" ") : (m.desc || ""),
   271‚Üí   271‚Üí    higher_level: Array.isArray(m.higher_level) ? m.higher_level.join(" ") : m.higher_level,
   272‚Üí   272‚Üí    classes: (m.classes || []).map((c: any) => c.name),
   273‚Üí   273‚Üí    subclasses: (m.subclasses || []).map((s: any) => s.name),
   274‚Üí   274‚Üí    kind,
   275‚Üí   275‚Üí    damage: dmgFormula,
   276‚Üí   276‚Üí    damageType: damage.damage_type?.name,
   277‚Üí   277‚Üí    damageScaling,
   278‚Üí   278‚Üí    saveAbility: hasSave ? ABILITY_MAP[(m.save.dc_type?.name || "").toLowerCase()] : undefined,
   279‚Üí   279‚Üí    saveSuccess: hasSave ? (m.save.dc_success === "half" ? "half" : "none") : undefined,
   280‚Üí   280‚Üí    heal: healFormula,
   281‚Üí   281‚Üí    aoeType: m.area_of_effect?.type,
   282‚Üí   282‚Üí    aoeSize: m.area_of_effect?.size,
   283‚Üí   283‚Üí    attackType: m.attack_type,
   284‚Üí   284‚Üí    conditionsAdd: conditionsAdd.length > 0 ? conditionsAdd : undefined,
   285‚Üí   285‚Üí    bonusAction,
   286‚Üí   286‚Üí    isCantrip: m.level === 0,
   287‚Üí   287‚Üí  };
   288‚Üí   288‚Üí}

   290‚Üí   290‚Üíexport async function fetchSpell(index: string, slotLevel?: number, charLevel = 1): Promise<NormalizedSpell | null> {
   291‚Üí   291‚Üí  const idx = String(index).toLowerCase().replace(/_/g, "-").trim();
   292‚Üí   292‚Üí  const raw = await fetchSRD(`spell=${encodeURIComponent(idx)}`);
   293‚Üí   293‚Üí  if (!raw) return null;
   294‚Üí   294‚Üí  return normalizeSpell(raw, slotLevel, charLevel);
   295‚Üí   295‚Üí}

   297‚Üí   297‚Üíexport async function fetchMonster(indexRaw: string): Promise<NormalizedMonster | null> {
   298‚Üí   298‚Üí  const index = String(indexRaw).toLowerCase().replace(/_/g, "-").trim();
   299‚Üí   299‚Üí  const m = await fetchSRD(`monster=${encodeURIComponent(index)}`);
   300‚Üí   300‚Üí  if (!m) return null;
   301‚Üí   301‚Üí
   302‚Üí   302‚Üí  // Parse actions into attack list
   303‚Üí   303‚Üí  const attacks: { name: string; atk: number; dmg: string; dmgType?: string }[] = [];
   304‚Üí   304‚Üí  for (const act of m.actions || []) {

   777‚Üí   777‚Üíexport function convertSRDMonsterToDefinition(m: NormalizedMonster): MonsterDefinition {
   778‚Üí   778‚Üí  const sizeMap: Record<string, CreatureSize> = {
   779‚Üí   779‚Üí    Tiny: "tiny",
   780‚Üí   780‚Üí    Small: "small",
   781‚Üí   781‚Üí    Medium: "medium",
   782‚Üí   782‚Üí    Large: "large",
   783‚Üí   783‚Üí    Huge: "huge",
   784‚Üí   784‚Üí    Gargantuan: "gargantuan",
   785‚Üí   785‚Üí  };
   786‚Üí   786‚Üí  const alignMap: Record<string, Alignment> = {
   787‚Üí   787‚Üí    "lawful good": "lawful_good",
   788‚Üí   788‚Üí    "neutral good": "neutral_good",
   789‚Üí   789‚Üí    "chaotic good": "chaotic_good",
   790‚Üí   790‚Üí    "lawful neutral": "lawful_neutral",
   791‚Üí   791‚Üí    "neutral": "true_neutral",
   792‚Üí   792‚Üí    "true neutral": "true_neutral",
   793‚Üí   793‚Üí    "chaotic neutral": "chaotic_neutral",
   794‚Üí   794‚Üí    "lawful evil": "lawful_evil",
   795‚Üí   795‚Üí    "neutral evil": "neutral_evil",
   796‚Üí   796‚Üí    "chaotic evil": "chaotic_evil",
   797‚Üí   797‚Üí    "unaligned": "unaligned",
   798‚Üí   798‚Üí    "any alignment": "true_neutral",
   799‚Üí   799‚Üí  };
   800‚Üí   800‚Üí  const size = sizeMap[m.size ?? "Medium"] ?? "medium";
   801‚Üí   801‚Üí  const type = (m.type?.toLowerCase() ?? "humanoid") as CreatureType;
   802‚Üí   802‚Üí  const alignment = alignMap[(m.alignment ?? "unaligned").toLowerCase()] ?? "unaligned";
   803‚Üí   803‚Üí
   804‚Üí   804‚Üí  // Convert special abilities + actions + legendary actions into MonsterAbility[]
   805‚Üí   805‚Üí  const abilities: MonsterAbility[] = [];
   806‚Üí   806‚Üí  for (const sa of m.specialAbilities ?? []) {
   807‚Üí   807‚Üí    abilities.push({
   808‚Üí   808‚Üí      id: `ab_${sa.name.toLowerCase().replace(/[^a-z0-9]+/g, "_")}`,
   809‚Üí   809‚Üí      kind: "trait",
   810‚Üí   810‚Üí      name: sa.name,
   811‚Üí   811‚Üí      description: sa.desc,
   812‚Üí   812‚Üí      passive: true,
   813‚Üí   813‚Üí    });
   814‚Üí   814‚Üí  }
   815‚Üí   815‚Üí  for (const a of m.actions ?? []) {
   816‚Üí   816‚Üí    abilities.push({
   817‚Üí   817‚Üí      id: `ac_${a.name.toLowerCase().replace(/[^a-z0-9]+/g, "_")}`,
   818‚Üí   818‚Üí      kind: "feature",
   819‚Üí   819‚Üí      name: a.name,
   820‚Üí   820‚Üí      description: a.desc,
   821‚Üí   821‚Üí    });
   822‚Üí   822‚Üí  }
   823‚Üí   823‚Üí  for (const la of m.legendaryActions ?? []) {
   824‚Üí   824‚Üí    abilities.push({
   825‚Üí   825‚Üí      id: `la_${la.name.toLowerCase().replace(/[^a-z0-9]+/g, "_")}`,
   826‚Üí   826‚Üí      kind: "legendary",
   827‚Üí   827‚Üí      name: la.name,
   828‚Üí   828‚Üí      description: la.desc,
   829‚Üí   829‚Üí    });
   830‚Üí   830‚Üí  }
   831‚Üí   831‚Üí  // Attacks
   832‚Üí   832‚Üí  const actions: MonsterAction[] = m.attacks.map((a, i) => ({
   833‚Üí   833‚Üí    id: `atk_${i}`,
   834‚Üí   834‚Üí    name: a.name,
   835‚Üí   835‚Üí    type: "action",
   836‚Üí   836‚Üí    description: `${a.name}: +${a.atk} to hit, ${a.dmg} ${a.dmgType ?? ""} damage`,
   837‚Üí   837‚Üí  }));
   838‚Üí   838‚Üí
   839‚Üí   839‚Üí  const stats: MonsterStats = {
   840‚Üí   840‚Üí    hp: m.hp,
   841‚Üí   841‚Üí    maxHp: m.hp,
   842‚Üí   842‚Üí    ac: m.ac,
   843‚Üí   843‚Üí    speed: 30,
   844‚Üí   844‚Üí    abilities: {
   845‚Üí   845‚Üí      str: 10, dex: 10, con: 10, int: 10, wis: 10, cha: 10,
   846‚Üí   846‚Üí    },
   847‚Üí   847‚Üí    savingThrows: m.sv,
   848‚Üí   848‚Üí    skills: {},
   849‚Üí   849‚Üí    passivePerception: 10 + (m.sv.wis ?? 0),
   850‚Üí   850‚Üí    languages: m.languages?.split(",").map((s) => s.trim()) ?? [],
   851‚Üí   851‚Üí  };
   852‚Üí   852‚Üí
   853‚Üí   853‚Üí  // Default AI behavior based on type
   854‚Üí   854‚Üí  const behavior: AIBehavior = {
   855‚Üí   855‚Üí    pattern: type === "beast" || type === "monstrosity" ? "aggressive" : "tactical",
   856‚Üí   856‚Üí    priorities: ["attack", "defend"],
   857‚Üí   857‚Üí    fleeThreshold: 25,
   858‚Üí   858‚Üí    preferTarget: "nearest",
   859‚Üí   859‚Üí  };
   860‚Üí   860‚Üí
   861‚Üí   861‚Üí  return {
   862‚Üí   862‚Üí    base: {
   863‚Üí   863‚Üí      id: m.index,
   864‚Üí   864‚Üí      name: m.name,
   865‚Üí   865‚Üí      type,
   866‚Üí   866‚Üí      size,
   867‚Üí   867‚Üí      alignment,
   868‚Üí   868‚Üí      cr: m.cr,
   869‚Üí   869‚Üí      source: "srd",
   870‚Üí   870‚Üí    },
   871‚Üí   871‚Üí    stats,
   872‚Üí   872‚Üí    actions,
   873‚Üí   873‚Üí    abilities,
   874‚Üí   874‚Üí    behavior,
   875‚Üí   875‚Üí  };
   876‚Üí   876‚Üí}
